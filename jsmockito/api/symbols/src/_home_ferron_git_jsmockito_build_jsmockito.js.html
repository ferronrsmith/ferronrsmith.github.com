 13  * Main namespace.
 14  * @namespace
 15  *
 16  * <h2>Contents</h2>
 17  *
 18  * <ol>
 19  *  <li><a href="#1">Let's verify some behaviour!</a></li>
 20  *  <li><a href="#2">How about some stubbing?</a></li>
 21  *  <li><a href="#3">Matching Arguments</a></li>
 22  *  <li><a href="#4">Verifying exact number of invocations / at least once /
 23  *  never</a></li>
 24  *  <li><a href="#5">Matching the context ('this')</a></li>
 25  *  <li><a href="#6">Making sure interactions never happened on a mock</a></li>
 26  *  <li><a href="#7">Finding redundant invocations</a></li>
 27  * </ol>
 28  *
 29  * <p>In the following examples object mocking is done with Array as this is
 30  * well understood, although you probably wouldn't mock this in normal test
 31  * development.</p>
 32  *
 33  * <h2><a name="1">1. Let's verify some behaviour!</a></h2>
 34  *
 35  * <p>For an object (via a prototype):</p>
 36  * <pre>
 37  * //mock creation
 38  * var prototype = new Array();
 39  * var mockedArray = mock(prototype);
 40  *
 41  * //using mock object
 42  * mockedArray.push("one");
 43  * mockedArray.reverse();
 44  *
 45  * //using mock object
 46  * mockedArray.push("one");
 47  * mockedArray.reverse();
 48  *
 49  * //verification
 50  * verify(mockedArray).push("one");
 51  * verify(mockedArray).reverse();
 52  * </pre>
 53  *
 54  * <p>For an object (via a constructor):</p>
 55  * <pre>
 56  * //mock creation
 57  * //WARNING: The constructor function will be run with no arguments!
 58  * //  Don't use this with functions that change the state of the application, use
 59  * //  a prototype instead (as described above).
 60  * var mockedArray = mock(Array);
 61  *
 62  * //using mock object
 63  * mockedArray.push("one");
 64  * mockedArray.reverse();
 65  *
 66  * //verification
 67  * verify(mockedArray).push("one");
 68  * verify(mockedArray).reverse();
 69  * </pre>
 70  *
 71  * <p>For a function:</p>
 72  * <pre>
 73  * //mock creation
 74  * var mockedFunc = mockFunction();
 75  *
 76  * //using mock function
 77  * mockedFunc('hello world');
 78  * mockedFunc.call(this, 'foobar');
 79  * mockedFunc.apply(this, [ 'barfoo' ]);
 80  *
 81  * //verification
 82  * verify(mockedFunc)('hello world');
 83  * verify(mockedFunc)('foobar');
 84  * verify(mockedFunc)('barfoo');
 85  * </pre>
 86  *
 87  * <p>Once created a mock will remember all interactions.  Then you selectively
 88  * verify whatever interactions you are interested in.</p>
 89  *
 90  * <h2><a name="2">2. How about some stubbing?</a></h2>
 91  *
 92  * <p>For an object:</p>
 93  * <pre>
 94  * var mockedArray = mock(Array);
 95  *
 96  * //stubbing
 97  * when(mockedArray).slice(0).thenReturn('f');
 98  * when(mockedArray).slice(1).thenThrow('An exception');
 99  * when(mockedArray).slice(2).then(function() { return 1+2 });
100  *
101  * //the following returns "f"
102  * assertThat(mockedArray.slice(0), equalTo('f'));
103  *
104  * //the following throws exception 'An exception'
105  * var ex = undefined;
106  * try {
107  *   mockedArray.slice(1);
108  * } catch (e) {
109  *   ex = e;
110  * }
111  * assertThat(ex, equalTo('An exception');
112  *
113  * //the following invokes the stub method, which returns 3
114  * assertThat(mockedArray.slice(2), equalTo(3));
115  *
116  * //the following returns undefined as slice(999) was not stubbed
117  * assertThat(mockedArray.slice(999), typeOf('undefined'));
118  *
119  * //stubs can take multiple values to return in order (same for 'thenThrow' and 'then' as well)
120  * when(mockedArray).pop().thenReturn('a', 'b', 'c');
121  * assertThat(mockedArray.pop(), equalTo('a'));
122  * assertThat(mockedArray.pop(), equalTo('b'));
123  * assertThat(mockedArray.pop(), equalTo('c'));
124  * assertThat(mockedArray.pop(), equalTo('c'));
125  *
126  * //stubs can also be chained to return values in order
127  * when(mockedArray).unshift().thenReturn('a').thenReturn('b').then(function() { return 'c' });
128  * assertThat(mockedArray.unshift(), equalTo('a'));
129  * assertThat(mockedArray.unshift(), equalTo('b'));
130  * assertThat(mockedArray.unshift(), equalTo('c'));
131  * assertThat(mockedArray.unshift(), equalTo('c'));
132  *
133  * //stub matching can overlap, allowing for specific cases and defaults
134  * when(mockedArray).slice(3).thenReturn('abcde');
135  * when(mockedArray).slice(3, lessThan(0)).thenReturn('edcba');
136  * assertThat(mockedArray.slice(3, -1), equalTo('edcba'));
137  * assertThat(mockedArray.slice(3, 1), equalTo('abcde'));
138  * assertThat(mockedArray.slice(3), equalTo('abcde'));
139  *
140  * //can also verify a stubbed invocation, although this is usually redundant
141  * verify(mockedArray).slice(0);
142  * </pre>
143  *
144  * <p>For a function:</p>
145  * <pre>
146  * var mockedFunc = mockFunction();
147  *
148  * //stubbing
149  * when(mockedFunc)(0).thenReturn('f');
150  * when(mockedFunc)(1).thenThrow('An exception');
151  * when(mockedFunc)(2).then(function() { return 1+2 });
152  *
153  * //the following returns "f"
154  * assertThat(mockedFunc(0), equalTo('f'))
155  *
156  * //following throws exception 'An exception'
157  * mockedFunc(1);
158  * //the following throws exception 'An exception'
159  * var ex = undefined;
160  * try {
161  *   mockedFunc(1);
162  * } catch (e) {
163  *   ex = e;
164  * }
165  * assertThat(ex, equalTo('An exception');
166  *
167  * //the following invokes the stub method, which returns 3
168  * assertThat(mockedFunc(2), equalTo(3));
169  *
170  * //following returns undefined as mockedFunc(999) was not stubbed
171  * assertThat(mockedFunc(999), typeOf('undefined'));
172  *
173  * //stubs can take multiple values to return in order (same for 'thenThrow' and 'then' as well)
174  * when(mockedFunc)(3).thenReturn('a', 'b', 'c');
175  * assertThat(mockedFunc(3), equalTo('a'));
176  * assertThat(mockedFunc(3), equalTo('b'));
177  * assertThat(mockedFunc(3), equalTo('c'));
178  * assertThat(mockedFunc(3), equalTo('c'));
179  *
180  * //stubs can also be chained to return values in order
181  * when(mockedFunc)(4).thenReturn('a').thenReturn('b').then(function() { return 'c' });
182  * assertThat(mockedFunc(4), equalTo('a'));
183  * assertThat(mockedFunc(4), equalTo('b'));
184  * assertThat(mockedFunc(4), equalTo('c'));
185  * assertThat(mockedFunc(4), equalTo('c'));
186  *
187  * //stub matching can overlap, allowing for specific cases and defaults
188  * when(mockedFunc)(5).thenReturn('abcde')
189  * when(mockedFunc)(5, lessThan(0)).thenReturn('edcba')
190  * assertThat(mockedFunc(5, -1), equalTo('edcba'))
191  * assertThat(mockedFunc(5, 1), equalTo('abcde'))
192  * assertThat(mockedFunc(5), equalTo('abcde'))
193  *
194  * //can also verify a stubbed invocation, although this is usually redundant
195  * verify(mockedFunc)(0);
196  * </pre>
197  *
198  * <ul>
199  * <li>By default mocks return undefined from all invocations;</li>
200  * <li>Stubs can be overwritten;</li>
201  * <li>Once stubbed, the method will always return the stubbed value regardless
202  * of how many times it is called;</li>
203  * <li>Last stubbing is more important - when you stubbed the same method with
204  * the same (or overlapping) matchers many times.</li>
205  * </ul>
206  *
207  * <h2><a name="3">3. Matching Arguments</a></h2>
208  *
209  * <p>JsMockito verifies arguments using 
210  * <a href="http://jshamcrest.destaquenet.com/">JsHamcrest</a> matchers.
211  *
212  * <pre>
213  * var mockedArray = mock(Array);
214  * var mockedFunc = mockFunction();
215  *
216  * //stubbing using JsHamcrest
217  * when(mockedArray).slice(lessThan(10)).thenReturn('f');
218  * when(mockedFunc)(containsString('world')).thenReturn('foobar');
219  *
220  * //following returns "f"
221  * mockedArray.slice(5);
222  *
223  * //following returns "foobar"
224  * mockedFunc('hello world');
225  *
226  * //you can also use matchers in verification
227  * verify(mockedArray).slice(greaterThan(4));
228  * verify(mockedFunc)(equalTo('hello world'));
229  *
230  * //if not specified then the matcher is anything(), thus either of these
231  * //will match an invocation with a single argument
232  * verify(mockedFunc)();
233  * verify(mockedFunc)(anything());
234  * </pre>
235  *
236  * <ul>
237  * <li>If the argument provided during verification/stubbing is not a
238  * JsHamcrest matcher, then 'equalTo(arg)' is used instead;</li>
239  * <li>Where a function/method was invoked with an argument, but the stub or
240  * verification does not provide a matcher, then anything() is assumed;</li>
241  * <li>The reverse, however, is not true - the anything() matcher will
242  * <em>not</em> match an argument that was never provided.</li>
243  * </ul>
244  *
245  * <h2><a name="4">4. Verifying exact number of invocations / at least once /
246  * never</a></h2>
247  *
248  * <pre>
249  * var mockedArray = mock(Array);
250  * var mockedFunc = mockFunction();
251  *
252  * mockedArray.slice(5);
253  * mockedArray.slice(6);
254  * mockedFunc('a');
255  * mockedFunc('b');
256  *
257  * //verification of multiple matching invocations
258  * verify(mockedArray, times(2)).slice(anything());
259  * verify(mockedFunc, times(2))(anything());
260  *
261  * //the default is times(1), making these are equivalent
262  * verify(mockedArray, times(1)).slice(5);
263  * verify(mockedArray).slice(5);
264  * </pre>
265  *
266  * <h2><a name="5">5. Matching the context ('this')</a></h2>
267  * 
268  * Functions can be invoked with a specific context, using the 'call' or
269  * 'apply' methods. JsMockito mock functions (and mock object methods)
270  * will remember this context and verification/stubbing can match on it.
271  *
272  * <p>For a function:</p>
273  * <pre>
274  * var mockedFunc = mockFunction();
275  * var context1 = {};
276  * var context2 = {};
277  *
278  * when(mockedFunc).call(equalTo(context2), anything()).thenReturn('hello');
279  *
280  * mockedFunc.call(context1, 'foo');
281  * //the following returns 'hello'
282  * mockedFunc.apply(context2, [ 'bar' ]);
283  *
284  * verify(mockedFunc).apply(context1, [ 'foo' ]);
285  * verify(mockedFunc).call(context2, 'bar');
286  * </pre>
287  *
288  * <p>For object method invocations, the context is usually the object itself.
289  * But sometimes people do strange things, and you need to test it - so
290  * the same approach can be used for an object:</p>
291  * <pre>
292  * var mockedArray = mock(Array);
293  * var otherContext = {};
294  *
295  * when(mockedArray).slice.call(otherContext, 5).thenReturn('h');
296  *
297  * //the following returns 'h'
298  * mockedArray.slice.apply(otherContext, [ 5 ]);
299  *
300  * verify(mockedArray).slice.call(equalTo(otherContext), 5);
301  * </pre>
302  *
303  * <ul>
304  * <li>For mock functions, the default context matcher is anything();</li>
305  * <li>For mock object methods, the default context matcher is
306  * sameAs(mockObj).</li>
307  * </ul>
308  *
309  * <h2><a name="6">6. Making sure interactions never happened on a mock</a></h2>
310  * 
311  * <pre>
312  * var mockOne = mock(Array);
313  * var mockTwo = mock(Array);
314  * var mockThree = mockFunction();
315  * 
316  * //only mockOne is interacted with
317  * mockOne.push(5);
318  *
319  * //verify a method was never called
320  * verify(mockOne, never()).unshift('a');
321  * 
322  * //verify that other mocks were not interacted with
323  * verifyZeroInteractions(mockTwo, mockThree);
324  * </pre>
325  *
326  * <h2><a name="7">7. Finding redundant invocations</a></h2>
327  *
328  * <pre>
329  * var mockArray = mock(Array);
330  * 
331  * mockArray.push(5);
332  * mockArray.push(8);
333  *
334  * verify(mockArray).push(5);
335  *
336  * // following verification will fail
337  * verifyNoMoreInteractions(mockArray);
338  * </pre>
339  */

406    * Create a mock that proxies a real function or object.  All un-stubbed

420   contextCaptureFunction: function(defaultContext, handler) {
421     // generate a function with overridden 'call' and 'apply' methods
422     // and apply a default context when these are not used to supply
423     // one explictly.
424     var captureFunction = function() {
425       return captureFunction.apply(defaultContext,
426         Array.prototype.slice.call(arguments, 0));
427     };
428     captureFunction.call = function(context) {
429       return captureFunction.apply(context,
430         Array.prototype.slice.call(arguments, 1));
431     };
432     captureFunction.apply = function(context, args) {
433       return handler.apply(this, [ context, args||[] ]);
434     };
435     return captureFunction;
436   },

438   matchArray: function(matchers, array) {
439     if (matchers.length > array.length)
440       return false;
441     return !JsMockito.any(matchers, function(matcher, i) {
442       return !matcher.matches(array[i]);
443     });
444   },

447     return JsHamcrest.isMatcher(obj)? obj :
448       JsHamcrest.Matchers.equalTo(obj);
449   },

459     JsMockito.Verifiers[name].prototype = new JsMockito.Verifier;

461     JsMockito.extend(JsMockito.Verifiers[name].prototype, proto);

498     var result = [];
499     JsMockito.each(srcArray, function(elem, key) {
500       var val = callback(elem, key);
501       if (val != null)

508     var result = [];
509     JsMockito.each(srcArray, function(elem, key) {
510       if (callback(elem, key))
511         result.push(elem);
512     });
513     return result;
514   },

516   find: function(array, callback) {
517     for (var i = 0; i < array.length; i++)
518       if (callback(array[i], i))
519         return array[i];
520     return undefined;
521   },
522 
523   any: function(array, callback) {
524     return (JsMockito.find(array, callback) != undefined);
525   },
526 
527   propertyNames: function(object) {
528     if (Object.getOwnPropertyNames) {
529       var prototype = Object.getPrototypeOf(object);
530       // top-most prototype object should be ignored
531       if (!prototype)
532         return [];
533       var names = JsMockito.propertyNames(prototype);
534       var ownNames = Object.getOwnPropertyNames(object);
535       JsMockito.each(ownNames, function(name) {
536         if (!JsMockito.any(names, function(n) { n == name }))
537           names.push(name);
538       });
539       return names;
540     } else {
541       var props = [];
542       for (var name in object)
543         props.push(name);
544       return props;
545     }
546   },
547 
548   propertyDescriptor: function(object, name) {
549     if (!Object.getOwnPropertyDescriptor)
550       return undefined;
551 
552     var descriptor = Object.getOwnPropertyDescriptor(object, name)
553     if (descriptor != undefined)
554       return descriptor;
555     var prototype = Object.getPrototypeOf(object);
556     if (!prototype || !Object.getPrototypeOf(prototype))
557       return undefined;
558     return this.propertyDescriptor(prototype, name);
559   },
560 
561   defineProperty: function(object, name, description) {
562     if (!Object.defineProperty) {
563       object[name] = description.value;
564       return;
565     }
566     Object.defineProperty(object, name, description);
567   },
568 
569   overwriteProperty: function(object, name, description) {
570     JsMockito.defineProperty(object, name, {
571       enumerable: description.enumerable,
572       configurable: true,
573       writable: true,
574       value: description.value
575     });
576   }

582 /**
583  * Create a mockable and stubbable anonymous function.
584  *
585  * <p>Once created, the function can be invoked and will return undefined for
586  * any interactions that do not match stub declarations.</p>
587  *
588  * <pre>
589  * var mockFunc = JsMockito.mockFunction();
590  * JsMockito.when(mockFunc).call(anything(), 1, 5).thenReturn(6);
591  * mockFunc(1, 5); // result is 6
592  * JsMockito.verify(mockFunc)(1, greaterThan(2));
593  * </pre>
594  *
595  * @param funcName {string} The name of the mock function to use in messages
596  *   (defaults to 'func')
597  * @param delegate {function} The function to delegate unstubbed calls to
598  *   (optional)
599  * @return {function} an anonymous function
600  */
601 JsMockito.mockFunction = function(funcName, delegate) {
602   if (typeof funcName == 'function') {
603     delegate = funcName;
604     funcName = undefined;
605   }
606   funcName = funcName || 'func';
607   delegate = delegate || function() { };
608 
609   var stubMatchers = []
610   var interactions = [];
611 
612   var mockFunc = function() {
613     var args = [this];
614     args.push.apply(args, arguments);
615     interactions.push({args: args});
616 
617     var stubMatcher = JsMockito.find(stubMatchers, function(stubMatcher) {
618       return JsMockito.matchArray(stubMatcher[0], args);
619     });
620     if (stubMatcher == undefined)
621       return delegate.apply(this, arguments);
622     var stubs = stubMatcher[1];
623     if (stubs.length == 0)
624       return undefined;
625     var stub = stubs[0];
626     if (stubs.length > 1)
627       stubs.shift();
628     return stub.apply(this, arguments);
629   };
630 
631   mockFunc.prototype = delegate.prototype;
632 
633   JsMockito.defineProperty(mockFunc, '_jsMockitoStubBuilder', {
634     enumerable: false,
635     value: function(contextMatcher) {
636       var contextMatcher = contextMatcher || JsHamcrest.Matchers.anything();
637       return matcherCaptureFunction(contextMatcher, function(matchers) {
638         var stubMatch = [matchers, []];
639         stubMatchers.unshift(stubMatch);
640         return {
641           then: function() {
642             stubMatch[1].push.apply(stubMatch[1], arguments);
643             return this;
644           },
645           thenReturn: function() {
646             return this.then.apply(this,JsMockito.map(arguments, function(value) {
647               return function() { return value };
648             }));
649           },
650           thenThrow: function(exception) {
651             return this.then.apply(this,JsMockito.map(arguments, function(value) {
652               return function() { throw value };
653             }));
654           }
655         };
656       });
657     }
658   });
659 
660   JsMockito.defineProperty(mockFunc, '_jsMockitoVerifier', {
661     enumerable: false,
662     value: function(verifier, contextMatcher) {
663       var contextMatcher = contextMatcher || JsHamcrest.Matchers.anything();
664       return matcherCaptureFunction(contextMatcher, function(matchers) {
665         return verifier(funcName, interactions, matchers, matchers[0] != contextMatcher);
666       });
667     }
668   });
669 
670   JsMockito.defineProperty(mockFunc, '_jsMockitoMockFunctions', {
671     enumerable: false,
672     value: function() {

691 /**
692  * Create a mockable and stubbable objects.
693  *
694  * <p>A mock is created with a object prototype as an argument, or with a constructor
695  * (which will be called with no arguments to construct a prototype). Once
696  * created, the mock object will have all the same methods as the prototype. These
697  * methods, when invoked, will return undefined by default.</p>
698  *
699  * <p><b>Note: </b>If a constructor is provided, it <i>will be called with no
700  * arguments</i>. Be careful not to pass constructors that will modify the
701  * state of the application when invoked!. And if you're mocking objects that
702  * require arguments to the constructor then you <i>must</i> create an instance
703  * first and supply that instance as a prototype.</p>
704  *
705  * <p>Stub declarations may then be made for these methods to have them return
706  * useful values or perform actions when invoked.</p>
707  *
708  * <pre>
709  * MyObject = function() {
710  *   this.add = function(a, b) { return a + b }
711  * };
712  *
713  * var mockObj = JsMockito.mock(MyObject);
714  * mockObj.add(5, 4); // result is undefined
715  *
716  * JsMockito.when(mockFunc).add(1, 2).thenReturn(6);
717  * mockObj.add(1, 2); // result is 6
718  *
719  * JsMockito.verify(mockObj).add(1, greaterThan(2)); // ok
720  * JsMockito.verify(mockObj).add(1, equalTo(2)); // ok
721  * JsMockito.verify(mockObj).add(1, 4); // will throw an exception
722  * </pre>
723  *
724  * Alternatively, using a prototype:
725  *
726  * <pre>
727  * myPrototype = {
728  *   add: function(a, b) { return a + b }
729  * };
730  *
731  * var mockObj = JsMockito.mock(myPrototype);
732  * mockObj.add(5, 4); // result is undefined
733  * </pre>
734  *
735  * @param Obj {function} the constructor for the object to be mocked
736  * @return {object} a mock object
737  */
738 JsMockito.mock = function(Obj, delegate) {
739   delegate = delegate || {};
740 
741   var MockObject = function() { };
742   try {
743     MockObject.prototype = (typeof Obj === "function")? new Obj : Obj;
744   } catch (e) {
745     throw new Error("mocked constructor threw an exception (consider mocking an object instead of a constructor)");
746   }
747   MockObject.prototype.constructor = MockObject;
748 
749   var mockObject = new MockObject();
750   var stubBuilders = {};
751   var verifiers = {};
752   var mockFunctions = [];
753 
754   var contextMatcher = JsHamcrest.Matchers.sameAs(mockObject);
755 
756   var addMockMethod = function(name) {
757     var delegateMethod;
758     if (delegate[name] != undefined) {
759       delegateMethod = function() {
760         var context = (this == mockObject)? delegate : this;
761         return delegate[name].apply(context, arguments);
762       };
763     }
764     var mockFunc = JsMockito.mockFunction('obj.' + name, delegateMethod);
765     var desc = JsMockito.propertyDescriptor(mockObject, name);
766     var enumerable = desc? desc.enumerable : true;
767 
768     JsMockito.defineProperty(mockObject, name, {
769       enumerable: enumerable,
770       writable: true,
771       configurable: true,
772       value: mockFunc
773     });
774     stubBuilders[name] = {
775       enumerable: enumerable,
776       value: mockFunc._jsMockitoStubBuilder
777     };
778     verifiers[name] = {
779       enumerable: enumerable,
780       value: mockFunc._jsMockitoVerifier
781     };
782     mockFunctions.push(mockFunc);
783   };
784 
785   JsMockito.each(JsMockito.propertyNames(mockObject), function(name) {
786     if (name == 'constructor')
787       return;
788     if (typeof mockObject[name] === 'function')
789       addMockMethod(name);
790     else if (delegate[name] != undefined) {
791       if (Object.defineProperty) {
792         Object.defineProperty(mockObject, name, {
793           get: function() { return delegate[name] },
794           set: function(val) { delegate[name] = val }
795         });
796       } else {
797         mockObject[name] = delegate[name];
798       }
799     }
800   });
801 
802   for (var typeName in JsMockito.nativeTypes) {
803     if (mockObject instanceof JsMockito.nativeTypes[typeName].type) {
804       JsMockito.each(JsMockito.nativeTypes[typeName].methods, function(method) {
805         addMockMethod(method);
806       });
807     }
808   }
809 
810   JsMockito.defineProperty(mockObject, '_jsMockitoStubBuilder', {
811     enumerable: false,
812     value: function() {
813       var stubBuilderObj = new MockObject();
814       JsMockito.eachObject(stubBuilders, function(desc, name) {
815         JsMockito.overwriteProperty(stubBuilderObj, name, {
816           enumerable: desc.enumerable,
817           value: desc.value.call(this, contextMatcher)
818         });
819       });
820       return stubBuilderObj;
821     }
822   });
823 
824   JsMockito.defineProperty(mockObject, '_jsMockitoVerifier', {
825     enumerable: false,
826     value: function(verifier) {
827       var verifierObj = new MockObject();
828       JsMockito.eachObject(verifiers, function(desc, name) {
829         JsMockito.overwriteProperty(verifierObj, name, {
830           enumerable: desc.enumerable,
831           value: desc.value.call(this, verifier, contextMatcher)
832         });
833       });
834       return verifierObj;
835     }
836   });
837 
838   JsMockito.defineProperty(mockObject, '_jsMockitoMockFunctions', {
839     enumerable: false,
840     value: function() {
841       return mockFunctions;
842     }
843   });
844 
845   return mockObject;
846 };
